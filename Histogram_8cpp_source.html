22 int64_t get_bucket_count(int64_t value, int64_t subb_count, int64_t unit_mag) noexcept
23 {
24  auto smallest_untrackable_value = subb_count << unit_mag;
25  auto buckets_needed = 1;
26  while (smallest_untrackable_value <= value) {
27  if (smallest_untrackable_value > (numeric_limits<int64_t>::max() / 2)) {
28  return buckets_needed + 1;
29  }
30  smallest_untrackable_value <<= 1;
31  ++buckets_needed;
32  }
33  return buckets_needed;
34 }

36 int64_t init_sub_bucket_half_count_magnitude(int64_t significant_figures)
37 {
38  double largest_value_single_unit_res = 2 * pow(10, significant_figures);
39  auto subb_count_mag = static_cast<int64_t>(ceil(log(largest_value_single_unit_res) / log(2)));
40  return subb_count_mag > 1 ? subb_count_mag - 1 : 0;
41 }

44 HdrHistogram::HdrHistogram(int64_t lowest_value, int64_t highest_value, int64_t significant_figures)
45 : unit_magnitude{static_cast<int64_t>(floor(log(lowest_value) / log(2)))}
46 , sub_bucket_half_count_magnitude{init_sub_bucket_half_count_magnitude(significant_figures)}
47 , sub_bucket_count(pow(2, sub_bucket_half_count_magnitude + 1))
48 , sub_bucket_half_count{sub_bucket_count / 2}
49 , sub_bucket_mask{(sub_bucket_count - 1) << unit_magnitude}
50 , bucket_count{get_bucket_count(highest_value, sub_bucket_count, unit_magnitude)}
51 , counts_len{(bucket_count + 1) * (sub_bucket_count / 2)}
52 {
53  if (highest_value < 1) {
54  throw runtime_error{"min value must be greater than zero"};
55  }
56  if (significant_figures < 1 || significant_figures > 5) {
57  throw runtime_error{"significant figures must be between 1 and 5"};
58  }
59  counts_.resize(counts_len);
60 }

75 int64_t HdrHistogram::percentile(double percentile) const
76 {
77  auto count_at_percentile = get_target_count_at_percentile(percentile);
78  int64_t total = 0;
79  for (int64_t i = 0; i < counts_len; ++i) {
80  total += get_count_at_index(i);
81  if (total >= count_at_percentile) {
82  auto value_at_index = get_value_from_index(i);
83  return percentile ? get_highest_equivalent_value(value_at_index)
84  : get_lowest_equivalent_value(value_at_index);
85  }
86  }
87  return 0;
88 }

113 double HdrHistogram::mean() const noexcept
114 {
115  if (!total_count_) {
116  return 0.0;
117  }
118  double total{0.0};
119  auto iter_adapter = RecordedIteratorAdapter(*this);
120  const auto& end = iter_adapter.end();
121  for (auto itr = iter_adapter.begin(); itr != end; ++itr) {
122  total += itr.get_count_at_this_value() * hdr_median_equiv_value(itr->value_iterated_to);
123  }
124  return total / total_count_;
125 }

127 double HdrHistogram::stddev() const noexcept
128 {
129  if (!total_count_) {
130  return 0.0;
131  }
132  const auto mean_value = mean();
133  double geometric_dev_total{0.0};
134  for (const auto& item : RecordedIteratorAdapter(*this)) {
135  const auto dev = (hdr_median_equiv_value(item.value_iterated_to) * 1.0) - mean_value;
136  geometric_dev_total += (dev * dev) * item.count_added_in_this_iter_step;
137  }
138  return sqrt(geometric_dev_total / total_count_);
139 }

152 int64_t HdrHistogram::get_highest_equivalent_value(int64_t value) const noexcept
153 {
154  auto bucket_index = get_bucket_index(value);
155  auto sub_bucket_index = get_sub_bucket_index(value, bucket_index);
156 
157  auto lowest_equivalent_value = get_value_from_sub_bucket(bucket_index, sub_bucket_index);
158  if (sub_bucket_index >= sub_bucket_count) {
159  ++bucket_index;
160  }
161  auto size_of_equivalent_value_range = 1ul << (unit_magnitude + bucket_index);
162  auto next_non_equivalent_value = lowest_equivalent_value + size_of_equivalent_value_range;
163  return next_non_equivalent_value - 1;
164 }

226 {

237 int64_t HdrHistogram::get_value_from_index(int64_t index) const noexcept
238 {
239  int bucket_index = (index >> sub_bucket_half_count_magnitude) - 1;
240  auto sub_bucket_index = (index & (sub_bucket_half_count - 1)) + sub_bucket_half_count;
241  if (bucket_index < 0) {
242  sub_bucket_index -= sub_bucket_half_count;
243  bucket_index = 0;
244  }
245  return get_value_from_sub_bucket(bucket_index, sub_bucket_index);
246 }

248 int64_t HdrHistogram::counts_index(int bucket_index, int sub_bucket_index) const noexcept
249 {
250  // Calculate the index for the first entry in the bucket:
251  // (The following is the equivalent of ((bucket_index + 1) * subBucketHalfCount) ):
252  auto bucket_base_index = (bucket_index + 1) << sub_bucket_half_count_magnitude;
253  // Calculate the offset in the bucket:
254  auto offset_in_bucket = sub_bucket_index - sub_bucket_half_count;
255  // The following is the equivalent of
256  // ((sub_bucket_index - subBucketHalfCount) + bucketBaseIndex
257  return bucket_base_index + offset_in_bucket;
258 }
