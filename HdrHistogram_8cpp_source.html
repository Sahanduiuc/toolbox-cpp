19 std::int64_t get_bucket_count(std::int64_t value, std::int64_t subb_count,
20  std::int64_t unit_mag) noexcept
21 {
22  auto smallest_untrackable_value = subb_count << unit_mag;
23  auto buckets_needed = 1;
24  while (smallest_untrackable_value <= value) {
25  if (smallest_untrackable_value > (std::numeric_limits<std::int64_t>::max() / 2)) {
26  return buckets_needed + 1;
27  }
28  smallest_untrackable_value <<= 1;
29  ++buckets_needed;
30  }
31  return buckets_needed;
32 }

34 std::int64_t init_sub_bucket_half_count_magnitude(std::int64_t significant_figures)
35 {
36  double largest_value_single_unit_res = 2 * std::pow(10, significant_figures);
37  auto subb_count_mag = static_cast<std::int64_t>(
38  std::ceil(std::log(largest_value_single_unit_res) / std::log(2)));
39  return subb_count_mag > 1 ? subb_count_mag - 1 : 0;
40 }

48 HdrHistogram::HdrHistogram(std::int64_t lowest_value, std::int64_t highest_value,
49  std::int64_t significant_figures)
50 : unit_magnitude{static_cast<std::int64_t>(std::floor(std::log(lowest_value) / std::log(2)))}
51 , sub_bucket_half_count_magnitude{init_sub_bucket_half_count_magnitude(significant_figures)}
52 , sub_bucket_count(std::pow(2, sub_bucket_half_count_magnitude + 1))
53 , sub_bucket_half_count{sub_bucket_count / 2}
54 , sub_bucket_mask{(sub_bucket_count - 1) << unit_magnitude}
55 , bucket_count{get_bucket_count(highest_value, sub_bucket_count, unit_magnitude)}
56 , counts_len{(bucket_count + 1) * (sub_bucket_count / 2)}
57 {
58  if (highest_value < 1) {
59  throw std::runtime_error{"Min value must be 1 or greater"};
60  }
61  if (significant_figures < 1 || significant_figures > 5) {
62  throw std::runtime_error{"SignificantFigures value must be between 1 and 5"};
63  }
64  counts_.resize(counts_len);
65 }

67 bool HdrHistogram::record(std::int64_t value, int count)
68 {
69  auto counts_index = counts_index_for(value);
70  if (counts_len <= counts_index) {
71  return false;
72  }
73  counts_[counts_index] += count;
74  total_count_ += count;
75  min_value_ = std::min(min_value_, value);
76  max_value_ = std::max(max_value_, value);
77  return true;

80 std::int64_t HdrHistogram::percentile(double percentile) const noexcept
81 {
82  auto count_at_percentile = get_target_count_at_percentile(percentile);
83  std::int64_t total = 0;
84  for (std::int64_t i = 0; i < counts_len; ++i) {
85  total += get_count_at_index(i);
86  if (total >= count_at_percentile) {
87  auto value_at_index = get_value_from_index(i);
88  return percentile ? get_highest_equivalent_value(value_at_index)
89  : get_lowest_equivalent_value(value_at_index);
90  }
91  }
92  return 0;
93 }

118 double HdrHistogram::mean() const noexcept
119 {
120  if (!total_count_) {
121  return 0.0;
122  }
123  double total{0.0};
124  auto iter_adapter = RecordedIteratorAdapter(*this);
125  const auto& end = iter_adapter.end();
126  for (auto itr = iter_adapter.begin(); itr != end; ++itr) {
127  total += itr.get_count_at_this_value() * hdr_median_equiv_value(itr->value_iterated_to);
128  }
129  return total / total_count_;
130 }

132 double HdrHistogram::stddev() const noexcept
133 {
134  if (!total_count_) {
135  return 0.0;
136  }
137  const auto mean_value = mean();
138  double geometric_dev_total{0.0};
139  for (const auto& item : RecordedIteratorAdapter(*this)) {
140  const auto dev = (hdr_median_equiv_value(item.value_iterated_to) * 1.0) - mean_value;
141  geometric_dev_total += (dev * dev) * item.count_added_in_this_iter_step;
142  }
143  return std::sqrt(geometric_dev_total / total_count_);
144 }

157 std::int64_t HdrHistogram::get_highest_equivalent_value(std::int64_t value) const noexcept
158 {
159  auto bucket_index = get_bucket_index(value);
160  auto sub_bucket_index = get_sub_bucket_index(value, bucket_index);
161 
162  auto lowest_equivalent_value = get_value_from_sub_bucket(bucket_index, sub_bucket_index);
163  if (sub_bucket_index >= sub_bucket_count) {
164  ++bucket_index;
165  }
166  auto size_of_equivalent_value_range = 1ul << (unit_magnitude + bucket_index);
167  auto next_non_equivalent_value = lowest_equivalent_value + size_of_equivalent_value_range;
168  return next_non_equivalent_value - 1;
169 }

209 std::int64_t HdrHistogram::get_bucket_index(std::int64_t value) const noexcept
210 {
211  // Smallest power of 2 containing value
212  auto pow2ceiling = 64 - __builtin_clzll(value | sub_bucket_mask);

222 std::int64_t HdrHistogram::get_value_from_index(std::int64_t index) const noexcept
223 {
224  int bucket_index = (index >> sub_bucket_half_count_magnitude) - 1;
225  auto sub_bucket_index = (index & (sub_bucket_half_count - 1)) + sub_bucket_half_count;
226  if (bucket_index < 0) {
227  sub_bucket_index -= sub_bucket_half_count;
228  bucket_index = 0;
229  }
230  return get_value_from_sub_bucket(bucket_index, sub_bucket_index);
231 }

233 std::int64_t HdrHistogram::counts_index(int bucket_index, int sub_bucket_index) const noexcept
234 {
235  // Calculate the index for the first entry in the bucket:
236  // (The following is the equivalent of ((bucket_index + 1) * subBucketHalfCount) ):
237  auto bucket_base_index = (bucket_index + 1) << sub_bucket_half_count_magnitude;
238  // Calculate the offset in the bucket:
239  auto offset_in_bucket = sub_bucket_index - sub_bucket_half_count;
240  // The following is the equivalent of
241  // ((sub_bucket_index - subBucketHalfCount) + bucketBaseIndex
242  return bucket_base_index + offset_in_bucket;
243 }

245 std::int64_t HdrHistogram::hdr_size_of_equiv_value_range(std::int64_t value) const noexcept
246 {
247  auto bucket_index = get_bucket_index(value);
248  auto sub_bucket_index = get_sub_bucket_index(value, bucket_index);
