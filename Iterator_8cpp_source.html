100 HdrIterator& HdrIterator::operator++()
101 {
102  while (has_next()) {
103  count_at_this_value_ = hist_.get_count_at_index(current_index_);
104  if (fresh_sub_bucket_) {
105  total_count_to_current_index_ += count_at_this_value_;
106  value_to_index_ += count_at_this_value_ * get_value_iterated_to();
107  fresh_sub_bucket_ = false;
108  }
109  if (reached_iteration_level()) {
110  auto value_iterated_to = get_value_iterated_to();
111  current_iteration_value_.set(value_iterated_to);
112 
113  prev_value_iterated_to_ = value_iterated_to;
114  total_count_to_prev_index_ = total_count_to_current_index_;
115 
116  increment_iteration_level();
117 
118  return *this;
119  }
120 
121  increment_sub_bucket();
122  }
123  return *this;
124 }

228 void PercentileIterator::increment_iteration_level()
229 {
230  this->percentile_to_iterate_from_ = this->percentile_to_iterate_to_;
231  auto percentile_gap{100.0 - this->percentile_to_iterate_to_};
232  if (percentile_gap) {
233  auto half_distance = pow(2, (log(100 / percentile_gap) / log(2)) + 1);
234  auto percentile_reporting_ticks = percentile_ticks_per_half_distance_ * half_distance;
235  this->percentile_to_iterate_to_ += 100.0 / percentile_reporting_ticks;
236  }
