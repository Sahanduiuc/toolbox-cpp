150  // Chaining breaks the lifetime extension of the rvalue and since we register the address in the
151  // store in the constructor chaining methods need to return move, and move constructors also
152  // register overriding the pointer value in the store.
153  template <typename FromT, typename ToT>
154  Benchmark range(const FromT& from, const ToT& to) && noexcept
155  {
156  func_.start = from;
157  func_.end = to;
158  return std::move(*this);
159  }
160 
161  template <typename ValueT>
162  Benchmark value(const ValueT& val) && noexcept
163  {
164  func_.start = val;
165  // TODO: this should be a different type.
166  func_.end = val + 1;
167  return std::move(*this);
168  }

170  void run() override final
171  {
172  using namespace std::literals::chrono_literals;
173 
174  constexpr auto sleep = 2s;
175 
176  // Warmup get clock timings.
177  // Measure single call.
178  std::atomic_bool done{false};
179  std::thread t([&]() {
180  std::this_thread::sleep_for(sleep);
181  done = true;
182  });
183  int runs{0};
184  while (!done) {
185  func_();
186  runs++;
187  }
188  t.join();
189 
190  const long long expected_seconds{3};
191  const long long n_calls = expected_seconds * runs / sleep.count();
192  auto start_check = std::chrono::high_resolution_clock::now();
193  for (int i = 0; i < n_calls; ++i) {
194  func_();
195  }
196  auto end_check = std::chrono::high_resolution_clock::now();
197  auto end_diff
198  = std::chrono::duration_cast<std::chrono::nanoseconds>(end_check - start_check);
199  std::cout << name_ << ' ' << end_diff.count() / (n_calls * func_.calls_per_execution())
200  << ' ' << n_calls << ' '
201  << std::chrono::duration_cast<std::chrono::microseconds>(end_diff).count()
202  << '\n';
203  }
204 
205  const char* name_;
206 
207  BenchmarkCaller<FuncT> func_;
208 };
