153  // Chaining breaks the lifetime extension of the rvalue and since we register the address in the
154  // store in the constructor chaining methods need to return move, and move constructors also
155  // register overriding the pointer value in the store.
156  template <typename FromT, typename ToT>
157  Benchmark range(const FromT& from, const ToT& to) && noexcept
158  {
159  func_.start = from;
160  func_.end = to;
161  return std::move(*this);
162  }

173  void run() final
174  {
175  using namespace std::literals::chrono_literals;
176 
177  constexpr auto sleep = 2s;
178 
179  // Warmup get clock timings.
180  // Measure single call.
181  std::atomic_bool done{false};
182  std::thread t([&]() {
183  std::this_thread::sleep_for(sleep);
184  done = true;
185  });
186  int runs{0};
187  while (!done) {
188  func_();
189  runs++;
190  }
191  t.join();
192 
193  const long long expected_seconds{3};
194  const long long n_calls = expected_seconds * runs / sleep.count();
195  auto start_check = std::chrono::high_resolution_clock::now();
196  for (int i = 0; i < n_calls; ++i) {
197  func_();
198  }
199  auto end_check = std::chrono::high_resolution_clock::now();
200  auto end_diff
201  = std::chrono::duration_cast<std::chrono::nanoseconds>(end_check - start_check);
202  std::cout << name_ << ' ' << end_diff.count() / (n_calls * func_.calls_per_execution())
203  << ' ' << n_calls << ' '
204  << std::chrono::duration_cast<std::chrono::microseconds>(end_diff).count()
205  << '\n';
206  }
207 
208  const char* name_;
209 
210  BenchmarkCaller<FuncT> func_;
211 };
