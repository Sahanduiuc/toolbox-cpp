116 AbstractHdrIterator& AbstractHdrIterator::operator++()
117 {
118  while (has_next()) {
119  count_at_this_value_ = histogram_.get_count_at_index(current_index_);
120  if (fresh_sub_bucket_) {
121  total_count_to_current_index_ += count_at_this_value_;
122  value_to_index_ += count_at_this_value_ * get_value_iterated_to();
123  fresh_sub_bucket_ = false;
124  }
125  if (reached_iteration_level()) {
126  auto value_iterated_to = get_value_iterated_to();
127  current_iteration_value_.set(value_iterated_to);
128 
129  prev_value_iterated_to_ = value_iterated_to;
130  total_count_to_prev_index_ = total_count_to_current_index_;
131 
132  increment_iteration_level();
133 
134  return *this;
135  }
136 
137  increment_sub_bucket();
138  }
139  return *this;
140 }

210 void PercentileIterator::increment_iteration_level()
211 {
212  this->percentile_to_iterate_from_ = this->percentile_to_iterate_to_;
213  auto percentile_gap{100.0 - this->percentile_to_iterate_to_};
214  if (percentile_gap) {
215  auto half_distance = std::pow(2, (std::log(100 / percentile_gap) / std::log(2)) + 1);
216  auto percentile_reporting_ticks = percentile_ticks_per_half_distance_ * half_distance;
217  this->percentile_to_iterate_to_ += 100.0 / percentile_reporting_ticks;
218  }
219 }
